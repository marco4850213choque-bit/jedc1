import { initializeApp } from "gstatic.com";  
import { getDatabase, ref, onValue, set, runTransaction } from "gstatic.com"; 

// !!! REEMPLAZA ESTO CON TUS DATOS DE FIREBASE !!!  
const firebaseConfig = {  
apiKey: "TU\_API\_KEY",  
authDomain: "TU\_firebaseapp.com",  
databaseURL: "https://TU\_firebaseio.com",  
projectId: "TU\_PROYECTO",  
storageBucket: "TU\_appspot.com"  
}; 

// Inicializar Firebase  
const app = initializeApp(firebaseConfig);  
const db = getDatabase(app);  
const puntosRef = ref(db, 'partida/puntos'); 

// Elementos de la interfaz  
const viewSeleccion = document.getElementById('view-seleccion');  
const viewBase = document.getElementById('view-base');  
const viewControl = document.getElementById('view-control');  
const scoreBase = document.getElementById('score-base');  
const scoreControl = document.getElementById('score-control'); 

// Escuchar cambios en la base de datos en tiempo real  
onValue(puntosRef, (snapshot) => {  
const puntos = snapshot.val() || 0;  
scoreBase.innerText = puntos;  
scoreControl.innerText = puntos;  
}); 

// Funciones para cambiar de pantalla  
window.mostrarBase = function() {  
viewSeleccion.classList.add('hidden');  
viewBase.classList.remove('hidden');  
} 

window.mostrarControl = function() {  
viewSeleccion.classList.add('hidden');  
viewControl.classList.remove('hidden');  
} 

// Funciones para sumar y restar puntos de forma segura (Transacciones)  
window.modificarPuntos = function(valor) {  
runTransaction(puntosRef, (currentPuntos) => {  
return (currentPuntos || 0) + valor;  
});  
} 

window.reiniciarPuntos = function() {  
set(puntosRef, 0);  
}

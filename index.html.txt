const juegos = \["Juego 1", "Juego 2", "Juego 3", "Juego 4", "Juego 5"\];  
let puntuaciones = { "Juego 1": 0, "Juego 2": 0, "Juego 3": 0, "Juego 4": 0, "Juego 5": 0 };  
let peer, conn;  
let conexionesControles = \[\];

// --- LÓGICA DEL CENTRO BASE ---  
function iniciarBase() {  
document.getElementById('view-inicio').classList.add('hidden');  
document.getElementById('view-base').classList.remove('hidden');  
dibujarBase();

// Genera un código corto aleatorio de 4 letras  
const idAleatorio = Math.random().toString(36).substring(2, 6).toUpperCase();  
peer = new Peer(idAleatorio);

peer.on('open', (id) => {  
document.getElementById('txt-codigo-base').innerText = id;  
});

// Escucha cuando un control remoto se conecta  
peer.on('connection', (c) => {  
conexionesControles.push(c);  
// Enviar las puntuaciones actuales al nuevo control  
c.on('open', () => c.send({ tipo: 'actualizar', datos: puntuaciones }));

// Escuchar si el control suma o resta  
c.on('data', (data) => {  
if (data.tipo === 'modificar') {  
puntuaciones\[data.juego\] += data.valor;  
dibujarBase();  
notificarATodos();  
} else if (data.tipo === 'reiniciar') {  
reiniciarTodo();  
}  
});  
});  
}

function dibujarBase() {  
const lista = document.getElementById('lista-juegos-base');  
lista.innerHTML = '';  
juegos.forEach(j => {  
lista.innerHTML += `<div class="game-row"> <span class="game-name">${j}</span> <span class="score">${puntuaciones[j]}</span> </div>`;  
});  
}

function reiniciarTodo() {  
juegos.forEach(j => puntuaciones\[j\] = 0);  
dibujarBase();  
notificarATodos();  
}

function notificarATodos() {  
conexionesControles.forEach(c => {  
if(c.open) c.send({ tipo: 'actualizar', datos: puntuaciones });  
});  
}

// --- LÓGICA DEL CONTROL REMOTO ---  
function iniciarControl() {  
const codigo = document.getElementById('input-codigo').value.trim().toUpperCase();  
if (!codigo) return alert("Escribe el código de la base primero");

peer = new Peer();  
peer.on('open', () => {  
conn = peer.connect(codigo);

conn.on('open', () => {  
document.getElementById('view-inicio').classList.add('hidden');  
document.getElementById('view-control').classList.remove('hidden');

// Llenar el selector  
const select = document.getElementById('select-juego');  
select.innerHTML = '';  
juegos.forEach(j => select.innerHTML += `<option value="${j}">${j}</option>`);  
});

// Recibir puntuaciones actualizadas desde la base  
conn.on('data', (data) => {  
if (data.tipo === 'actualizar') {  
puntuaciones = data.datos;  
actualizarScoreControl();  
}  
});  
});  
}

function actualizarScoreControl() {  
const juegoSel = document.getElementById('select-juego').value;  
document.getElementById('score-control').innerText = puntuaciones\[juegoSel\];  
}

function enviarPuntos(valor) {  
const juegoSel = document.getElementById('select-juego').value;  
conn.send({ tipo: 'modificar', juego: juegoSel, valor: valor });  
}
